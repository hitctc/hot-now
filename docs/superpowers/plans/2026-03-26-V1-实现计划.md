# HotNow V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local single-process service that fetches the `juya-ai-daily` RSS feed, clusters linked articles into deterministic `Top 10` topics, stores daily JSON/HTML reports, shows latest/history pages, and emails the report through QQ SMTP.

**Architecture:** A TypeScript Fastify app owns one run pipeline shared by scheduler and manual trigger. The pipeline fetches RSS, parses the daily issue, grabs linked articles, clusters topics with explicit rules, renders JSON/HTML artifacts under a per-day directory, updates web pages from stored files, and sends email without blocking report generation on SMTP success.

**Tech Stack:** Node.js 20, TypeScript, Fastify, rss-parser, cheerio, jsdom + @mozilla/readability, node-cron, nodemailer, Vitest, tsx

// src/core/config/loadRuntimeConfig.ts
import { readFile } from "node:fs/promises";
import path from "node:path";
import { RuntimeConfig } from "../types/appConfig.js";

type Options = {
  configPath?: string;
  env?: Record<string, string | undefined>;
};

export async function loadRuntimeConfig(options: Options = {}): Promise<RuntimeConfig> {
  const env = options.env ?? process.env;
  const configPath = options.configPath ?? path.resolve("config/hot-now.config.json");
  const fileText = await readFile(configPath, "utf8");
  const fileConfig = JSON.parse(fileText) as Omit<RuntimeConfig, "smtp">;

  return {
    ...fileConfig,
    report: {
      ...fileConfig.report,
      dataDir: path.resolve(path.dirname(configPath), fileConfig.report.dataDir)
    },
    smtp: {
      host: required(env.SMTP_HOST, "SMTP_HOST"),
      port: Number(required(env.SMTP_PORT, "SMTP_PORT")),
      secure: required(env.SMTP_SECURE, "SMTP_SECURE") === "true",
      user: required(env.SMTP_USER, "SMTP_USER"),
      pass: required(env.SMTP_PASS, "SMTP_PASS"),
      to: required(env.MAIL_TO, "MAIL_TO"),
      baseUrl: required(env.BASE_URL, "BASE_URL")
    }
  };
}

function required(value: string | undefined, key: string) {
  if (!value) {
    throw new Error(`Missing required env: ${key}`);
  }

  return value;
}

// src/core/storage/reportStore.ts
import { mkdir, readdir, readFile, writeFile } from "node:fs/promises";
import path from "node:path";

export function reportDayDir(rootDir: string, reportDate: string) {
  return path.join(rootDir, reportDate);
}

export async function writeJsonFile(rootDir: string, reportDate: string, fileName: string, data: unknown) {
  const dayDir = reportDayDir(rootDir, reportDate);
  await mkdir(dayDir, { recursive: true });
  await writeFile(path.join(dayDir, fileName), JSON.stringify(data, null, 2));
}

export async function writeTextFile(rootDir: string, reportDate: string, fileName: string, content: string) {
  const dayDir = reportDayDir(rootDir, reportDate);
  await mkdir(dayDir, { recursive: true });
  await writeFile(path.join(dayDir, fileName), content, "utf8");
}

export async function readTextFile(rootDir: string, reportDate: string, fileName: string) {
  return readFile(path.join(reportDayDir(rootDir, reportDate), fileName), "utf8");
}

export async function readJsonFile<T>(rootDir: string, reportDate: string, fileName: string): Promise<T> {
  return JSON.parse(await readTextFile(rootDir, reportDate, fileName)) as T;
}

export async function listReportDates(rootDir: string) {
  try {
    return (await readdir(rootDir, { withFileTypes: true }))
      .filter((entry) => entry.isDirectory())
      .map((entry) => entry.name)
      .sort()
      .reverse();
  } catch {
    return [];
  }
}

export async function listReportSummaries(rootDir: string) {
  const dates = await listReportDates(rootDir);
  return Promise.all(
    dates.map(async (date) => {
      const report = await readJsonFile<any>(rootDir, date, "report.json");
      return {
        date,
        topicCount: report.meta.topicCount,
        degraded: report.meta.degraded,
        mailStatus: report.meta.mailStatus
      };
    })
  );
}

// src/core/source/parseJuyaIssue.ts
import Parser from "rss-parser";
import * as cheerio from "cheerio";

export type CandidateItem = {
  rank: number;
  category: string;
  title: string;
  sourceUrl: string;
};

export type DailyIssue = {
  date: string;
  issueUrl: string;
  items: CandidateItem[];
};

const parser = new Parser();

export async function parseJuyaIssue(feedXml: string): Promise<DailyIssue> {
  const feed = await parser.parseString(feedXml);
  const latestItem = feed.items[0];

  if (!latestItem) {
    throw new Error("RSS feed does not contain any issue items");
  }

  const date = latestItem.title ?? "unknown-date";
  const issueUrl = latestItem.link ?? "";
  const html = latestItem["content:encoded"] ?? latestItem.content ?? "";
  const $ = cheerio.load(html);
  const items: CandidateItem[] = [];
  let currentCategory = "未分类";

  $("h3, li").each((_, element) => {
    if (element.tagName === "h3") {
      currentCategory = $(element).text().trim();
      return;
    }

    const link = $(element).find("a").attr("href") ?? "";
    const rankText = $(element).find("code").text().trim();
    const rank = Number(rankText.replace("#", ""));
    const title = $(element)
      .clone()
      .find("a, code")
      .remove()
      .end()
      .text()
      .replace("↗", "")
      .trim();

    if (link && Number.isFinite(rank) && title) {
      items.push({ rank, category: currentCategory, title, sourceUrl: link });
    }
  });

  return { date, issueUrl, items };
}

import { JSDOM } from "jsdom";
import { Readability } from "@mozilla/readability";

export type ArticleResult =
  | { ok: true; url: string; title: string; text: string; siteName?: string }
  | { ok: false; url: string; title: string; text: string; error: string };

export async function extractArticle(url: string, html: string): Promise<ArticleResult> {
  try {
    const dom = new JSDOM(html, { url });
    const parsed = new Readability(dom.window.document).parse();

    if (!parsed?.textContent?.trim()) {
      return {
        ok: false,
        url,
        title: parsed?.title ?? "",
        text: "",
        error: "Readable text not found"
      };
    }

    return {
      ok: true,
      url,
      title: parsed.title ?? "",
      text: parsed.textContent.trim().slice(0, 6000),
      siteName: parsed.siteName
    };
  } catch (error) {
    return {
      ok: false,
      url,
      title: "",
      text: "",
      error: error instanceof Error ? error.message : "Unknown extraction error"
    };
  }
}

export async function fetchAndExtractArticle(url: string): Promise<ArticleResult> {
  const response = await fetch(url);

  if (!response.ok) {
    return { ok: false, url, title: "", text: "", error: `HTTP ${response.status}` };
  }

  return extractArticle(url, await response.text());
}
```

// src/core/topics/clusterTopics.ts
type RankedInput = {
  rank: number;
  category: string;
  title: string;
  sourceUrl: string;
  article: { ok: boolean; url: string; title: string; text: string };
};

export type Topic = {
  topicKey: string;
  score: number;
  category: string;
  headline: string;
  items: RankedInput[];
};

const PRIORITY_BONUS: Record<string, number> = {
  "要闻": 30,
  "模型发布": 20,
  "开发生态": 15
};

export function clusterTopics(items: RankedInput[]): Topic[] {
  const buckets = new Map<string, Topic>();

  for (const item of items) {
    const topicKey = deriveTopicKey(item.title);
    const existing = buckets.get(topicKey);
    const score = 100 - item.rank + (PRIORITY_BONUS[item.category] ?? 0);

    if (!existing) {
      buckets.set(topicKey, {
        topicKey,
        score,
        category: item.category,
        headline: item.title,
        items: [item]
      });
      continue;
    }

    existing.score += score + 10;
    existing.items.push(item);
    if (item.rank < existing.items[0].rank) {
      existing.headline = item.title;
      existing.category = item.category;
    }
  }

  return [...buckets.values()].sort((left, right) => right.score - left.score);
}

function deriveTopicKey(title: string) {
  const normalized = title.toLowerCase();

  if (normalized.includes("lyria")) return "lyria";
  if (normalized.includes("claude")) return "claude";
  if (normalized.includes("figma")) return "figma";

  return normalized
    .replace(/[^\p{L}\p{N}\s]/gu, " ")
    .split(/\s+/)
    .filter(Boolean)
    .slice(0, 2)
    .join("-");
}
```

export function summarizeTopic(topic: Topic) {
  const bestItem = topic.items.slice().sort((left, right) => left.rank - right.rank)[0];
  const body = bestItem.article.text || bestItem.title;

  return {
    title: topic.headline,
    category: topic.category,
    whyItMatters: `该主题当前聚合了 ${topic.items.length} 条候选信息，且最高源内排序为 #${bestItem.rank}。`,
    summary: body.slice(0, 140),
    keywords: extractKeywords(topic.headline),
    relatedCount: topic.items.length
  };
}

function extractKeywords(title: string) {
  return title
    .replace(/[^\p{L}\p{N}\s]/gu, " ")
    .split(/\s+/)
    .filter((token) => token.length >= 2)
    .slice(0, 5);
}

// src/core/runtime/runLock.ts
export function createRunLock() {
  let running = false;

  return {
    isRunning() {
      return running;
    },
    async runExclusive<T>(task: () => Promise<T>) {
      if (running) {
        throw new Error("A digest run is already in progress");
      }

      running = true;
      try {
        return await task();
      } finally {
        running = false;
      }
    }
  };
}

// src/core/pipeline/runDailyDigest.ts
import { buildDailyReport } from "../report/buildDailyReport.js";
import { renderReportHtml } from "../report/renderReportHtml.js";
import { writeJsonFile, writeTextFile } from "../storage/reportStore.js";
import { clusterTopics } from "../topics/clusterTopics.js";

export async function runDailyDigest(config: any, trigger: "manual" | "scheduled", deps: any) {
  const issue = await deps.loadLatestIssue(config);
  const enrichedItems = await Promise.all(
    issue.items.map(async (item: any) => ({
      ...item,
      article: await deps.fetchArticle(item.sourceUrl)
    }))
  );

  const topics = clusterTopics(enrichedItems);
  const report = buildDailyReport({ issue, trigger, topics, topN: config.report.topN });

  let mailStatus = "sent";
  try {
    await deps.sendDailyEmail(config, report);
  } catch (error) {
    mailStatus = error instanceof Error ? `failed:${error.message}` : "failed:unknown";
  }

  report.meta.mailStatus = mailStatus;
  const html = renderReportHtml(report);

  await writeJsonFile(config.report.dataDir, report.meta.date, "report.json", report);
  await writeTextFile(config.report.dataDir, report.meta.date, "report.html", html);

  await writeJsonFile(config.report.dataDir, report.meta.date, "run-meta.json", {
    trigger,
    generatedAt: report.meta.generatedAt,
    mailStatus,
    degraded: report.meta.degraded
  });

  return { report, mailStatus };
}

// src/server/createServer.ts
import Fastify from "fastify";
import { renderControlPage, renderHistoryPage } from "./renderPages.js";

type Deps = {
  config?: any;
  listReportSummaries?: () => Promise<Array<{ date: string; topicCount: number; degraded: boolean; mailStatus: string }>>;
  latestReportDate?: () => Promise<string | null>;
  readReportHtml?: (date: string) => Promise<string>;
  triggerManualRun?: () => Promise<{ accepted: boolean }>;
  isRunning?: () => boolean;
};

export function createServer(deps: Deps = {}) {
  const app = Fastify({ logger: true });

  app.get("/health", async () => ({ ok: true }));

  app.get("/", async (_request, reply) => {
    const latestDate = (await deps.latestReportDate?.()) ?? null;
    if (!latestDate) {
      return reply.type("text/html").send("<h1>今日尚未生成报告</h1>");
    }

    return reply.type("text/html").send(await deps.readReportHtml?.(latestDate));
  });

  app.get("/history", async (_request, reply) => {
    const summaries = (await deps.listReportSummaries?.()) ?? [];
    return reply.type("text/html").send(renderHistoryPage(summaries));
  });

  app.get("/reports/:date", async (request: any, reply) => {
    return reply.type("text/html").send(await deps.readReportHtml?.(request.params.date));
  });

  app.get("/control", async (_request, reply) => {
    return reply.type("text/html").send(renderControlPage(deps.config, deps.isRunning?.() ?? false));
  });

  app.post("/actions/run", async (_request, reply) => {
    if (deps.isRunning?.()) {
      return reply.code(409).send({ accepted: false, reason: "already-running" });
    }

    if (!deps.triggerManualRun) {
      return reply.code(503).send({ accepted: false });
    }

    const result = await deps.triggerManualRun();
    return reply.code(202).send(result);
  });

  return app;
}

import { createServer } from "./server/createServer.js";
import { loadRuntimeConfig } from "./core/config/loadRuntimeConfig.js";
import { listReportDates, listReportSummaries, readTextFile } from "./core/storage/reportStore.js";
import { loadLatestIssue } from "./core/source/loadLatestIssue.js";
import { fetchAndExtractArticle } from "./core/fetch/extractArticle.js";
import { sendDailyEmail } from "./core/mail/sendDailyEmail.js";
import { runDailyDigest } from "./core/pipeline/runDailyDigest.js";
import { createRunLock } from "./core/runtime/runLock.js";
import { startScheduler } from "./core/scheduler/startScheduler.js";

const config = await loadRuntimeConfig();
const lock = createRunLock();

async function trigger(triggerType: "manual" | "scheduled") {
  return lock.runExclusive(() =>
    runDailyDigest(config, triggerType, {
      loadLatestIssue,
      fetchArticle: fetchAndExtractArticle,
      sendDailyEmail
    })
  );
}

const app = createServer({
  config,
  isRunning: () => lock.isRunning(),
  listReportSummaries: () => listReportSummaries(config.report.dataDir),
  latestReportDate: async () => (await listReportDates(config.report.dataDir))[0] ?? null,
  readReportHtml: (date: string) => readTextFile(config.report.dataDir, date, "report.html"),
  triggerManualRun: async () => {
    await trigger("manual");
    return { accepted: true };
  }
});

startScheduler(config, async () => {
  await trigger("scheduled");
});

await app.listen({ host: "127.0.0.1", port: config.server.port });
